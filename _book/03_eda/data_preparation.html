#acces aux tables  sans chargement en memoire
GetLazyDBAccess <- function(conn,tables){
  lazyDb <- lapply(tables, function(tbl_name) {
    tbl(conn, tbl_name) # Accesses each table from the connection
  })
  lazyDb
}
